128 /** \brief returns the list of parent, parent-of-parent, ... that can be cast to type \a T */
129 template<class T>
130 inline QList<T*> getParents() {
131 QList<T*> lst;
132 JKQTMathTextNode* p=getParentNode();
133 T* pT=dynamic_cast<T*>(p);
134 while (p!=nullptr) {
135 if (pT!=nullptr) lst.append(pT);
136 p=p->getParentNode();
137 pT=dynamic_cast<T*>(p);
138 }
139 return lst;
140 }
141
142 /** \brief returns the list of parent, parent-of-parent, ... that can be cast to type \a T */
143 template<class T>
144 inline QList<const T*> getParents() const {
145 QList<const T*> lst;
146 const JKQTMathTextNode* p=getParentNode();
147 const T* pT=dynamic_cast<const T*>(p);
148 while (p!=nullptr) {
149 if (pT!=nullptr) lst.append(pT);
150 p=p->getParentNode();
151 pT=dynamic_cast<const T*>(p);
152 }
153 return lst;
154 }
