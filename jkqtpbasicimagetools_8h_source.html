51/*! \brief retrieve an R/G/B/Alpha (\a ch == 0/1/2/3) value from the \c QRgb value \a rgb
52 \ingroup jkqtplotter_imagelots_tools
53 */
54inline int JKQTPGetColorChannel(QRgb rgb, int ch)
55{
56 switch(ch) {
57 case 0: return qRed(rgb);
58 case 1: return qGreen(rgb);
59 case 2: return qBlue(rgb);
60 case 3: return qAlpha(rgb);
61 }
62 return qGray(rgb);
63}
64
65/*! \brief set the R/G/B/Alpha (\a ch == 0/1/2/3) value in the \c QRgb value \a rgb to \a val (0..255!)
66 \ingroup jkqtplotter_imagelots_tools
67 */
68inline void JKQTPSetColorChannel(QRgb& rgb, int ch, int val)
69{
70 switch(ch) {
71 case 0: rgb= qRgba(val, qGreen(rgb), qBlue(rgb), qAlpha(rgb)); break;
72 case 1: rgb= qRgba(qRed(rgb), val, qBlue(rgb), qAlpha(rgb)); break;
73 case 2: rgb= qRgba(qRed(rgb), qGreen(rgb), val, qAlpha(rgb)); break;
74 case 3: rgb= qRgba(qRed(rgb), qGreen(rgb), qBlue(rgb), val); break;
75 }
76}

288 /*! \brief convert a 2D image (as 1D array) into a QImage with given palette (see JKQTFPColorPalette)
289
290 \param dbl_in pointer to a 1D array of template type \c T representing the image to plot. This array has to be of size \a width * \a height
291 \param width width of the array in \a dbl
292 \param height height of the array in \a dbl
293 \param[out] img the QImage object to draw to (should be initialized as \c QImage::Format_ARGB32 )
294 \param palette the color palette to use for the display
295 \param minColor lower boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
296 \param maxColor upper boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
297 \param paletteMinFail specifies what shell happen, when a value in \a dbl is below \a minColor
298 \param paletteMaxFail specifies what shell happen, when a value in \a dbl is above \a maxColor
299 \param minFailColor color to use for pixels that are below \a minColor for some settings of \a paletteMinFail
300 \param maxFailColor color to use for pixels that are below \a maxColor for some settings of \a paletteMaxFail
301 \param nanColor color to use for pixels that are not-a-number
302 \param infColor color to use for pixels that are infinity
303 \param logScale create a log-scaled image
304 \param logBase base for the logarithm used when \c logScale==true
305 \param lutUser user define LUT, used if \a palette \c ==JKQTPMathImageUSER_PALETTE
306 \note There is a variant of this function that is called with a userLUT directly, instead of \a palette
307 */
308 template <class T>
309 static inline void array2image(const T* dbl_in, int width, int height, QImage &img, JKQTPMathImageColorPalette palette, double minColor, double maxColor, JKQTPMathImageColorRangeFailAction paletteMinFail=JKQTPMathImageLastPaletteColor, JKQTPMathImageColorRangeFailAction paletteMaxFail=JKQTPMathImageLastPaletteColor, QColor minFailColor=QColor("black"), QColor maxFailColor=QColor("black"), QColor nanColor=QColor("black"), QColor infColor=QColor("black"), bool logScale=false, double logBase=10.0, const LUTType& lutUser=LUTType())
310 {
311 if (!dbl_in || width<=0 || height<=0)
312 return;
313
314 double min = *dbl_in;
315 double max = *dbl_in;
316 if (jkqtp_approximatelyEqual(minColor, maxColor, JKQTP_DOUBLE_EPSILON)) {
317 bool first=true;
318 for (int i=1; i<width*height; ++i)
319 {
320 T v=dbl_in[i];
321 if (!(std::isnan(static_cast<long double>(v)) || std::isinf(static_cast<long double>(v)))) {
322 if (first) {
323 min=max=v;
324 first=false;
325 } else {
326 if (v < min)
327 min = v;
328 else if (v > max)
329 max = v;
330 }
331 }
332 }
333
334 } else {
335 min = minColor;
336 max = maxColor;
337 }
338
339 const T* dbl=dbl_in;
340 QVector<T> dbl1;
341 if (logScale) {
342 double logB=log10(logBase);
343 dbl1=QVector<T>(width*height, 0);
344 for (int i=0; i<width*height; i++) {
345 dbl1[i]=log10(dbl_in[i])/logB;
346 }
347 dbl=dbl1.data();
348 min=log10(min)/logB;
349 max=log10(max)/logB;
350 }
351 double delta=max-min;
352
353
354 // either use lutUser (if palette==JKQTPMathImageUSER_PALETTE) or else read the LUT from the internal storage
355 const LUTType& lut_used=(palette==JKQTPMathImageUSER_PALETTE)?lutUser:getLUTforPalette(JKQTPImageTools::global_jkqtpimagetools_lutstore, palette);
356
357 img = QImage(width, height, QImage::Format_ARGB32);
358 if (jkqtp_approximatelyEqual(min, max, JKQTP_DOUBLE_EPSILON)) {
359 img.fill(0);
360 } else {
361 const int lutSize=lut_used.size();
362 if (lutSize>0) {
363 // LUT found: collor the image accordingly
364 for (int j=0; j<height; ++j) {
365 QRgb* line=reinterpret_cast<QRgb*>(img.scanLine(height-1-j));
366 for (int i=0; i<width; ++i) {
367 double val=dbl[j*width+i];
368 if (std::isnan(val)) {
369 line[i]=nanColor.rgba();
370 } else if (std::isinf(val)) {
371 line[i]=infColor.rgba();
372 } else {
373 const int v = static_cast<int>((val-min)/delta*static_cast<double>(lutSize-1));
374 const int vv = (v < 0) ? 0 : ( (v >= lutSize) ? (lutSize-1) : v);
375 line[i]=lut_used[vv];
376 if ((v<0)&&(paletteMinFail==JKQTPMathImageGivenColor)) {
377 line[i]=minFailColor.rgba();
378 } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageGivenColor)) {
379 line[i]=maxFailColor.rgba();
380 } else if ((v<0)&&(paletteMinFail==JKQTPMathImageTransparent)) {
381 line[i]=QColor(Qt::transparent).rgba();
382 } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageTransparent)) {
383 line[i]=QColor(Qt::transparent).rgba();
384 }
385 }
386 }
387 }
388 } else {
389 // no LUT found: paint a black image!
390 img.fill(0);
391 }
392 }
