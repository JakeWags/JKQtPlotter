55/*! \brief represents a font specifier for JKQTMathText. The font consists of two parts: the actual font and the font used for math output (which may be empty)
56 \ingroup jkqtmathtext
57
58 \section JKQTMathTextFontSpecifier_specialNames Special FOnt Names
59 This object also implements replacing special font names with actual fonts. Supported special font names are:
60 - \c default / \c app / \c application - the applications default font
61 - \c times / \c serif - a general serif font
62 - \c sans-serif - a general sans-serif font
63 - \c typewriter - a general typewrter/monospaced font
64 - \c cursive
65 - \c decorative
66 - \c fantasy
67 - \c monospace
68 - \c system
69 .
70
71 If copiled with Qt>5.3 you can also use these:
72 - \c fixed
73 - \c smallest_readable
74 - \c title
75 - \c general
76 .
77*/

82 static JKQTMathTextFontSpecifier fromFontSpec(const QString& fontSpec);
83
84 /** \brief initialises the object with values from parsing a \a fontSpec string with the form \c "FONT_NAME[+MATH_FONT_NAME]". */
85 void setFontSpec(const QString& fontSpec);

88 QString getFontSpec() const;

90 QString fontName() const;
91 /** \copydoc m_mathFontName */
92 QString mathFontName() const;
93
94 /** \copydoc m_fontName */
95 void setFontName(const QString& name);
96 /** \copydoc m_mathFontName */
97 void setmathFontName(const QString& name);
98 /** \brief finds actual fonts for some predefined special font names, as listed in \ref JKQTMathTextFontSpecifier_specialNames */
99 static QString transformFontName(const QString& fontName);
100 /** \brief same as transformFontName(), but also finds the actual name for XITS, STIX, ASANA,... */
101 static QString transformFontNameAndDecodeSpecialFonts(const QString& fontName);
102 /** \brief leiefert \c true, wenn ein fontName() verfügbar ist */
