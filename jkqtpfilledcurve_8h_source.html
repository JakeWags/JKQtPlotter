33/** \brief a Base class for filled curve graphs like e.g. JKQTPFilledCurveXGraph
34 * \ingroup jkqtplotter_linesymbolgraphs_simple

36 * \image html filledgraphs.png

38 * \see JKQTPFilledCurveXGraph, JKQTPFilledCurveYGraph

40class JKQTPLOTTER_LIB_EXPORT JKQTPFilledCurveGraphBase: public JKQTPXYBaselineGraph, public JKQTPGraphLineAndFillStyleMixin {
41 Q_OBJECT
42public:
43 /** \brief specifies how the area below the graph is filled
44 *
45 * \see setFillMode(), getFillMode(), fillStyleBelow(), \ref JKQTPlotterWigglePlots
46 */
47 enum FillMode {
48 SingleFilling=0, /*!< \brief the whole area is filled with the same color/pattern, generates "simple filled plots", such as \image html filledgraphs_small.png */
49 TwoColorFilling=1 /*!< \brief the area above and below baseline with the two different colors/patterns, generates "wiggle plots", such as \image html wiggleplots_small.png */
50 };
51 Q_ENUM(FillMode)
52
53 /** \brief class constructor */
54 explicit JKQTPFilledCurveGraphBase(JKQTBasePlotter* parent=nullptr);
55
56 /** \brief returns the color to be used for the key label */
57 virtual QColor getKeyLabelColor() const override;
58 /** \brief plots a key marker inside the specified rectangle \a rect */
59 virtual void drawKeyMarker(JKQTPEnhancedPainter& painter, QRectF& rect) override;
60 /** \copydoc m_fillStyleBelow */
61 JKQTPGraphFillStyleMixin& fillStyleBelow();
62 /** \copydoc m_fillStyleBelow */
63 const JKQTPGraphFillStyleMixin& fillStyleBelow() const;
64 /** \copydoc m_fillMode */
65 FillMode getFillMode() const;
66public slots:
67 /** \brief set line-color, fill color and symbol color */
68 void setColor(QColor c);
69 /** \copydoc m_fillMode */
70 void setFillMode(FillMode mode);
71protected:
72 /** \brief specifies how the area of the graph is filles */
73 FillMode m_fillMode;
74 /** \brief if m_fillMode \c ==FillAboveAndBelowDifferently then this fill style is used below the baseline and
75 * the default fill style is used above */
76 JKQTPGraphFillStyleMixin m_fillStyleBelow;
77};

81/*! \brief This implements filled curve plots where the area is filled between the plot line and the x-Axis.
82 \ingroup jkqtplotter_filledgraphs
83
84 \image html filledgraphs.png
85
86 \see \ref JKQTPlotterFilledGraphs
87
88 This class also provides the possibility to file above and below the baseline with different style.
89 Such plots are sometimes called "Wiggle Plots" and are often used (in their black/white-variety for seismographic
90 data plotting.
91
92 \image html wiggleplot_x.png
93
94 To generate such a plot, used \c JKQTPFilledCurveXGraph::setFillMode(JKQTPFilledCurveXGraph::FillMode::TwoColorFilling)
95 and then use \c JKQTPFilledCurveXGraph::fillStyleBelow() to access the fill style for the area below the baseline
96 and the default fill style to define the fill style above the baseline.
97
98 \see \ref JKQTPlotterWigglePlots
99
100 */
101class JKQTPLOTTER_LIB_EXPORT JKQTPFilledCurveXGraph: public JKQTPFilledCurveGraphBase {
102 Q_OBJECT
103 public:
104 /** \brief class constructor */
105 JKQTPFilledCurveXGraph(JKQTBasePlotter* parent=nullptr);
106 /** \brief class constructor */
107 JKQTPFilledCurveXGraph(JKQTPlotter* parent);
108 /** \brief plots the graph to the plotter object specified as parent */
109 void draw(JKQTPEnhancedPainter &painter);
110};
